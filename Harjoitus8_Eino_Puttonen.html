#Timestamp-sarakkeesta vuosi ja kuukausi
df['Timestamp'] = pd.to_datetime(df['Timestamp'])
df['Year'] = df['Timestamp'].dt.year
df['Month'] = df['Timestamp'].dt.month
#tehdään yksi kuva missä on kaksi subplotia
plt.figure(figsize=(13, 5))
#määritetään rivien ja sarakkaiden määrä, lopuksi määritetään ensimmäin subplot
plt.subplot(1, 2, 1)
# Piirakkakaavio hyökkäystyyppien prosentuaalisista osuuksista
attack_type_counts = df['Attack Type'].value_counts()
plt.pie(attack_type_counts, labels=attack_type_counts.index, autopct='%1.1f%%')
plt.title('Hyökkäystyyppien prosentuaaliset osuudet datasta')
plt.axis('equal')
# Pylväskaavio
plt.subplot(1, 2, 2) 
malware_attacks_2022 = df[(df['Attack Type'] == 'Malware') & (df['Year'] == 2022)]
monthly_malware_counts = malware_attacks_2022.groupby('Month').size()
monthly_malware_counts.plot(kind='bar', color='deepskyblue', width=0.7
                           )
plt.title('Hyökkäystyypin "Malware" lukumäärät kuukausittain vuonna 2022')
plt.xlabel('Kuukausi')
plt.ylabel('Hyökkäysten lukumäärä')
plt.xticks(rotation=0)



plt.show()
